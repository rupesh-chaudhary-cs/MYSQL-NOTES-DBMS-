# Experiment 9: Subqueries

## Aim
To perform subquery operations.

---

## Queries

```sql
-- 1. Employee with highest salary.
SELECT ENAME FROM EMPLOYEE
WHERE SAL = (SELECT MAX(SAL) FROM EMPLOYEE);

-- 2. Highest paid clerk.
SELECT EMPNO, ENAME FROM EMPLOYEE
WHERE JOB='CLERK' AND SAL=(SELECT MAX(SAL) FROM EMPLOYEE WHERE JOB='CLERK');

-- 3. Salesman earning more than any clerk.
SELECT ENAME FROM EMPLOYEE
WHERE JOB='SALESMAN'
AND SAL > (SELECT MAX(SAL) FROM EMPLOYEE WHERE JOB='CLERK');

-- 4. Clerks salary condition.
SELECT ENAME FROM EMPLOYEE
WHERE JOB='CLERK'
AND SAL > (SELECT SAL FROM EMPLOYEE WHERE ENAME='JAMES')
AND SAL < (SELECT SAL FROM EMPLOYEE WHERE ENAME='SCOTT');

-- 5. Employees salary > James or > Scott.
SELECT ENAME FROM EMPLOYEE
WHERE SAL > (SELECT SAL FROM EMPLOYEE WHERE ENAME='JAMES')
OR SAL > (SELECT SAL FROM EMPLOYEE WHERE ENAME='SCOTT');

-- 6. Highest salary dept-wise.
SELECT ENAME FROM EMPLOYEE E
WHERE SAL = (SELECT MAX(SAL) FROM EMPLOYEE WHERE DEPTNO=E.DEPTNO);

-- 7. Highest salary job-wise.
SELECT ENAME FROM EMPLOYEE E
WHERE SAL = (SELECT MAX(SAL) FROM EMPLOYEE WHERE JOB=E.JOB);

-- 8. Employees in accounting.
SELECT ENAME FROM EMPLOYEE
WHERE DEPTNO = (SELECT DEPTNO FROM DEPARTMENT WHERE DNAME='ACCOUNTING');

-- 9. Employees in Chicago (assumed dept mapping).
SELECT ENAME FROM EMPLOYEE;

-- 10. Job groups salary greater than managers.
SELECT JOB FROM EMPLOYEE
GROUP BY JOB
HAVING SUM(SAL) > (SELECT MAX(SAL) FROM EMPLOYEE WHERE JOB='MANAGER');
